\tBeginVerbatim{}

\tEndVerbatim

\tBeginVerbatim{ title = to test trimming }
\clozeset[ green ]{text_color=green}
\tEndVerbatim

\tBeginVerbatim{ title = global options }
\clozeset[]{text_color=yellow}
\tEndVerbatim

\tBeginVerbatim{}
Default global group (text color is yellow): \cloze{yellow}
\tEndVerbatim

\tBeginVerbatim{}
Group green: \cloze[group=green]{green}
\tEndVerbatim

\tBeginVerbatim{}
Group green specified with spaces \cloze[group= green ]{green}
\tEndVerbatim

\tBeginVerbatim{}
Group red: \cloze[group=red]{red}
\tEndVerbatim

\tAll

\tAssertAllGroupOpts{
  green =
  {
    text_color=green
  },
  red =
  {
    text_color=red
  }
}

\tAssertGlobalOpts{text_color=yellow}

\tAssertGroupOpts{green}{text_color=green}

\tAssertGroupOpts{red}{text_color=red}
