\input cloze.tex

\font\TestTypewriterFontBigger=cmtt10 scaled \magstep3
\font\TestTypewriterFontBig=cmtt10 scaled \magstep1
\font\TestTypewriterFontNormal=cmtt10

\def\TestTitle#1{%
  \noindent%
  {\TestTypewriterFontBigger #1}%
}

\def\TestDescription#1{%
  \noindent%
  {\TestTypewriterFontNormal #1}%
}

\def\TestTitleDesc#1#2{%
  \TestTitle{#1}
  \par
  \TestDescription{#2}
  \par
  {\noindent\TestTypewriterFontNormal Test file: \jobname.tex}
  \par
  \hrule width 6cm
  \bigskip
}

\def\TestSection#1{%
  \bigskip%
  \noindent%
  {\TestTypewriterFontBig#1}%
  \bigskip%
}

\def\TestShowHideText#1#2{%
  \par%
  \bigskip\bigskip%
  \noindent%
  {\TestTypewriterFontNormal Cloze texts are #1 (visibility=#2):}%
  \par\bigskip%
}

\def\TestShowCloze{%
  \clozeshow%
  \TestShowHideText{displayed}{true}%
}

\def\TestHideCloze{%
  \clozehide%
  \TestShowHideText{hidden}{false}%
}

\def\TestShowAndHide{
  \TestShowCloze

  \TestExample

  \TestHideCloze

  \TestExample
}
