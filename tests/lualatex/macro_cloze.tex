\section{Behaviour after punctation}

\def\clozepunctation{%
  Lorem ipsum dolor sit amet, consectetur adipisicing elit sit amet
  \cloze{Lorem}.
  \cloze{ipsum},
  \cloze{dolor};
  \cloze{sit}: %
}

\def\Repeat[#1]#2{
  \count0=#1
  \loop
  \ifnum\count0>0
  \advance\count0 by -1
  #2%
  \repeat
}
\Repeat[20]\clozepunctation
