\def\createcatcodes{
  \bgroup
    \catcode`\\=12
    \catcode`\{=12
    \catcode`\}=12
    \catcode`\$=12
    \catcode`\&=12
    \catcode`\^^M=13
    \catcode`\#=12
    \catcode`\^=12
    \catcode`\_=12
    \catcode`\ =13
    \catcode`\~=12
    \catcode`\%=12
    \savecatcodetable 1
  \egroup
}
\createcatcodes

\def\Space{ }

  \catcode`\^^M=13\gdef^^M{\quitvmode\par}%
