\directlua{
   cloze_test = require('cloze-test')
}

\input cloze.tex

\def\createcatcodes{
  \bgroup
    \catcode`\\=12
    \catcode`\{=12
    \catcode`\}=12
    \catcode`\$=12
    \catcode`\&=12
    \catcode`\^^M=13
    \catcode`\#=12
    \catcode`\^=12
    \catcode`\_=12
    \catcode`\ =13
    \catcode`\~=12
    \catcode`\%=12
    \savecatcodetable 1
  \egroup
}
\createcatcodes

\def\Space{ }
\bgroup
  \catcode`\^^M=13\gdef^^M{\quitvmode\par}%
  \catcode`\ = 13\gdef {\quitvmode\Space}%
\egroup

\def\ClozeTest{%
  \directlua{cloze_test.register_verbatim()}%
}

\def\EndClozeTest{}

\def\useverbatim{%
   \directlua{cloze_test.print_lines()}%
}

\def\printverbatim{%
   \bgroup\parindent=0pt \tt
   \directlua{cloze_test.print_lines(1)}
   \egroup
}
